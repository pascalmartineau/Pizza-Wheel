  <script>

    const wheel = document.getElementById("wheel");

    const spinBtn = document.getElementById("spinBtn");

    const result = document.getElementById("result");

    const names = ["Sophia", "Courtney", "Sophia", "Courtney", "Sophia", "Courtney"];

    const sliceAngle = 60;

    let currentRotation = 0;

    let spinning = false;

    function spinWheel() {

      if (spinning) return;

      spinning = true;

      spinBtn.disabled = true;

      result.textContent = "";

      const winningIndex = Math.floor(Math.random() * 6);

      // Pointer is at top. Slice centers are at:

      // 30, 90, 150, 210, 270, 330 degrees

      // We rotate so the winning slice center lands at 0/top.

      const targetSliceCenter = winningIndex * sliceAngle + 30;

      const extraSpins = 360 * (5 + Math.floor(Math.random() * 3)); // 5-7 spins

      const finalRotation = extraSpins + (360 - targetSliceCenter);

      currentRotation += finalRotation;

      wheel.style.transform = `rotate(${currentRotation}deg)`;

      setTimeout(() => {

        const winner = names[winningIndex];

        result.textContent = `🤌 The pizza chooses: ${winner}!`;

        spinBtn.disabled = false;

        spinning = false;

      }, 4900);

    }

    spinBtn.addEventListener("click", spinWheel);

  </script>
